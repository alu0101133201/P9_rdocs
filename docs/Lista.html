def extraer_cabeza

  if(@head != nil)

    x = @head
    
    @head = x.next
    x.next = nil
    @head.prev = nil

      @tail = @head

def extraer_cola

  if(@tail != nil)

    x = @tail
    
    @tail = x.prev
    x.prev = nil
    @tail.next = nil

    if(@head == nil)
      @head = @tail
    end

  return x.value

  end

  return @tail

end

    Convierte la lista a string

    # File lib/Pract6/lista.rb, line 24
def to_s

  string = "" 

  if(@head == nil)
    return string
  end

  x = @head

  while x.next != nil
    aux = x.value
    string << "#{aux},"
    x = x.next
  end

  string << "#{x.value}"

return string 


end
